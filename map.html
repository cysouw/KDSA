# read coordinates
coor <- read.delim("data/coor.txt", row.names = 1)

# prepare map

library(qlcVisualize)

window <- hullToOwin(coor, shift = 0.1, alpha = 0.2)
v <- voronoi(coor, window)
## 
##      PLEASE NOTE:  The components "delsgs" and "summary" of the
##  object returned by deldir() are now DATA FRAMES rather than
##  matrices (as they were prior to release 0.0-18).
##  See help("deldir").
##  
##      PLEASE NOTE: The process that deldir() uses for determining
##  duplicated points has changed from that used in version
##  0.0-9 of this package (and previously). See help("deldir").
vmap(v)
## Warning in rep(col, length.out = length(tiles)): 'x' is NULL so the result
## will be NULL

save(v, file = "data/KDSAvoronoi.Rdata")
